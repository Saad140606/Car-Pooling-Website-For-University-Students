# Campus Ride - Stops System Implementation Guide

## Quick Start

### 1. Database Schema Update (Firestore)

Update your ride document to include stops:

```firestore
universities/{univId}/rides/{rideId}
{
  // ... existing fields
  
  stops: [{
    id: string (unique ID)
    name: string (human readable)
    lat: number (latitude)
    lng: number (longitude)
    distanceFromStart: number (meters from route start)
    type: string (landmark, bus_stop, intersection, etc.)
    placeId: string (OSM reference)
    isCustom: boolean (true if user-created)
    isAutoGenerated: boolean (true if system-generated)
    order: number (sequence position)
    createdAt: Timestamp
    modifiedAt: Timestamp
  }]
  
  stopsLastUpdated: Timestamp
}
```

### 2. Ride Creation Flow - Add Stops

Update your `create-ride/page.tsx`:

```typescript
'use client';

import { useStops } from '@/lib/useStops';
import StopsViewer from '@/components/StopsViewer';

export default function CreateRidePage() {
  const { generateStops, updateStops, loading } = useStops();
  const [stops, setStops] = useState<Stop[]>([]);

  // After user selects route and clicks "Next"
  const handleRouteSelected = async (route: any) => {
    // Generate automatic stops
    const generatedStops = await generateStops(
      route.polyline,
      route.distance,
      route.from,
      route.to
    );
    
    if (generatedStops) {
      setStops(generatedStops);
    }
  };

  // Before saving ride
  const handleCreateRide = async (rideData: any) => {
    // Create ride in Firestore
    const rideRef = await addDoc(...); // your ride creation logic
    
    // Save stops
    await updateStops(rideRef.id, stops);
  };

  return (
    <div>
      {/* ... route selection UI ... */}
      
      {stops.length > 0 && (
        <StopsViewer
          stops={stops}
          isCreator={true}
          onUpdateStops={(newStops) => setStops(newStops)}
          triggerText="Edit Stops"
        />
      )}
      
      <Button onClick={handleCreateRide}>
        Create Ride
      </Button>
    </div>
  );
}
```

### 3. Ride Viewer - Show View Stops Button

Update your ride card component:

```typescript
import StopsViewer from '@/components/StopsViewer';
import { useStops } from '@/lib/useStops';

export function RideCard({ ride }: { ride: any }) {
  const { getStops, loading } = useStops();
  const [stops, setStops] = useState<Stop[]>([]);

  useEffect(() => {
    const loadStops = async () => {
      const fetchedStops = await getStops(ride.id, ride.university);
      setStops(fetchedStops || []);
    };
    loadStops();
  }, [ride.id]);

  return (
    <div>
      {/* ... existing ride card UI ... */}
      
      <StopsViewer
        stops={stops}
        isCreator={false}
        triggerText="View Stops"
      />
    </div>
  );
}
```

## API Endpoints

### Generate Stops
```bash
POST /api/rides/generate-stops

Body:
{
  routePolyline: "string",
  routeDistance: 5000,
  from: "Location A",
  to: "Location B"
}

Response:
{
  success: true,
  stops: [
    {
      id: "stop_123",
      name: "Main Gate",
      lat: 24.8607,
      lng: 67.0011,
      distanceFromStart: 500,
      type: "landmark"
    }
  ]
}
```

### Get Ride Stops
```bash
GET /api/rides/{rideId}/stops?university=fast

Response:
{
  success: true,
  stops: [...]
}
```

### Update Stops
```bash
PUT /api/rides/{rideId}/stops
Authorization: Bearer {token}

Body:
{
  stops: [...],
  university: "fast"
}

Response:
{
  success: true,
  message: "Stops updated successfully",
  stops: [...]
}
```

### Add Custom Stop
```bash
POST /api/rides/{rideId}/stops
Authorization: Bearer {token}

Body:
{
  stop: {
    name: "Custom Stop",
    lat: 24.8607,
    lng: 67.0011,
    distanceFromStart: 1000,
    type: "custom"
  },
  university: "fast"
}

Response:
{
  success: true,
  stop: {...}
}
```

### Delete Stop
```bash
DELETE /api/rides/{rideId}/stops?stopId={stopId}&university=fast
Authorization: Bearer {token}

Response:
{
  success: true,
  message: "Stop deleted successfully"
}
```

## Component Usage

### StopsViewer Component

For ride creators to view/edit stops:

```typescript
import StopsViewer from '@/components/StopsViewer';

<StopsViewer
  stops={stops}
  routePolyline={ride.routePolyline}
  isCreator={true}
  onUpdateStops={async (newStops) => {
    await updateStops(rideId, newStops);
  }}
  triggerText="Manage Stops"
/>
```

**Props:**
- `stops`: Stop[] - Array of stops
- `routePolyline`: string (optional) - Route for map display
- `isCreator`: boolean - Show edit/delete/add buttons
- `onUpdateStops`: async function - Called when saving
- `triggerText`: string - Button text

**Features:**
- Drag-to-reorder stops
- Rename stops
- Add custom stops
- Delete stops
- Map preview

### StopSelector Component

For passengers to select pickup stop:

```typescript
import StopSelector from '@/components/StopSelector';

<StopSelector
  stops={ride.stops}
  onSelectStop={(stopId, stopName) => {
    setPickupStop(stopId);
  }}
  triggerText="Select Pickup Stop"
/>
```

**Props:**
- `stops`: Stop[] - Array of stops
- `onSelectStop`: (stopId, stopName) => void
- `triggerText`: string - Button text
- `selectedStopId`: string (optional) - Pre-selected stop

## Performance Considerations

1. **Caching**: Cache generated stops in client state to avoid regeneration
2. **Lazy Loading**: Load stops only when viewing ride details
3. **Stop Limit**: Cap stops at 10 per ride maximum
4. **API Rate Limiting**: Use debouncing when auto-generating
5. **Database Indexes**: Add Firestore composite index on `universities/{univId}/rides` for queries
